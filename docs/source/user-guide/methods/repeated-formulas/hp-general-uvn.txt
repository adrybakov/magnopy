  \begin{pmatrix} S^+ \\  S^- \\   S^n \end{pmatrix}

  \begin{pmatrix}S^u + iS^v\\ S^u - iS^v\\  S^n\end{pmatrix}

  \begin{pmatrix}
  \sqrt{2S}\,\,\sqrt{1 - \dfrac{a^{\dagger}a}{2S}}\,\,a
  \\
  \sqrt{2S} \,\, a^{\dagger} \,\,\sqrt{1 - \dfrac{a^{\dagger}a}{2S}}
  \\
   S\,\,(1 - \dfrac{a^{\dagger}\,a}{S})
  \end{pmatrix}
